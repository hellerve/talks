    \begin{listing}[H]
      \caption{A safer format}
      \begin{minted}{clojure}
(defmacro fmt [s & args]
  (let [a (dec (count (filter #(not= "" %)
                              (str/split s #"%"))))
        b (count args)]
    (if (= a b)
      `(apply format ~s '~args)
      (exc "arguments to format don’t match (expected " a
           ", got " b ")"))))
      \end{minted}
    \end{listing}
